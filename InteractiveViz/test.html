library(shiny)
  library(plotly)
library(plotly)
  
  # Sample data
  set.seed(123)
  df <- data.frame(
    x = rnorm(100),
    y = rnorm(100)
  )
  
  # Create scatterplot
  p <- plot_ly(df, x = ~x, y = ~y, type = 'scatter', mode = 'markers',
               marker = list(size = 10),
               hoverinfo = 'text',
               text = ~paste('X:', x, 'Y:', y)) # Custom hover text
  
  # Show plot
  p
# Define UI
  ui <- fluidPage(
    plotlyOutput("plot"),
    sliderInput("xrange", "X-Axis Range", min(df$x), max(df$x), value = range(df$x)),
    sliderInput("yrange", "Y-Axis Range", min(df$y), max(df$y), value = range(df$y))
  )
  
  # Define server logic
  server <- function(input, output) {
    output$plot <- renderPlotly({
      p <- plot_ly(df, x = ~x, y = ~y, type = 'scatter', mode = 'markers') %>%
        layout(xaxis = list(range = input$xrange),
               yaxis = list(range = input$yrange))
      p
    })
  }
  
  # Run the app
  shinyApp(ui = ui, server = server)
